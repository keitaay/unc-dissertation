% Set position and line margins for the ToC title

% Set title; visible heading in ALL CAPS, PDF metadata/bookmark in sentence case.
% The trailing "hfill" keeps the visible heading centered.
\renewcommand{\contentsname}{\texorpdfstring{\textbf{\MakeTextUppercase{Table of Contents}}\hfill}{Table of Contents}}
\renewcommand{\cfttoctitlefont}{\hfill}
\renewcommand{\cftaftertoctitle}{\hfill}

% Configure page numbers. Note that "cftsetrmarg" sets the 
% right margin. For example, "1.6in" is approximately 25 picas,
% or 75% of the full text width in this document (6.5in).
% This size will be used for the right margin of the page numbers.
\renewcommand{\cftdotsep}{1.5} % Density of dot leaders
\cftsetrmarg{1.0in}
\setlength{\cftchapindent}{0pt} % left margin of first line
\renewcommand{\cftchapleader}{\cftdotfill{\cftdotsep}}

% Set line spacings for each entry
\setlength{\cftbeforetoctitleskip}{61pt}
\setlength{\cftaftertoctitleskip}{12pt}
\setlength{\cftbeforechapskip}{15pt}
\setlength{\cftbeforesecskip}{10pt}
\setlength{\cftbeforesubsecskip}{10pt}
\setlength{\cftbeforesubsubsecskip}{10pt}

% Render table of contents
\begin{singlespace}
\tableofcontents
\end{singlespace}
