\setlength{\cftafterlottitleskip}{22pt}

\renewcommand{\cftafterlottitle}{\hfill}

% Set line spacings for each entry
\setlength{\cftbeforetabskip}{10pt}

% Left-align first line of entries, then
% indent wrapped lines by label width
\setlength{\cfttabindent}{0pt}
\setlength{\cfttabnumwidth}{2.8em}
\renewcommand{\cfttableader}{\cftdotfill{\cftdotsep}}
