% Use proper margins
\geometry{letterpaper,left=1in,top=1in,right=1in,bottom=1in,nohead}

% Double-space text
\doublespacing

% Format chapter titles:
% - Center chapter titles
% - Omit page numbers
\titleformat{\chapter}[display]{\fillast\bfseries}{\Large\MakeUppercase{\chaptertitlename} \thechapter}{-11pt}{\huge\singlespacing}[\thispagestyle{empty}]

% Extend top margins to 2in + double-space line break after heading
\titlespacing{\chapter}{0in}{0.62in}{2em}

% Indent paragraphs four spaces throughout the document
\setlength{\parindent}{4ex}

% Tweak spacing of paragraph labels
\titlespacing{\paragraph}{0in}{0.08in}{0.07in}

% We want numbered subsubsections
% \setcounter{secnumdepth}{3}
% \setcounter{tocdepth}{3}

% Double-space line breaks between footnotes
\setlength{\footnotesep}{\baselineskip}

% We don't want crazy vertical spacing
\raggedbottom

% We don't want abandoned words
\clubpenalty=10000 
\widowpenalty=10000

% Prevent awkward hyphenations
\hyphenation{Raj-kumar}

% Fully left-align footnotes
\makeatletter
\renewcommand\@makefntext[1]{%
  \noindent\makebox[7pt][r]{\@makefnmark\ }#1}
\makeatother

% Make footnotes same size as rest of text
\renewcommand{\footnotesize}{\normalsize}

% Ensure superscripts and subscripts are only slightly smaller
% compared to the font size of the main text. Note that the
% arguments define the base, normal text, subscript, and
% superscript font sizes, respectively.
\DeclareMathSizes{12}{12}{10}{10}

% Single space footnotes if they are longer than one line
\makeatletter
\renewcommand{\@makefntext}[1]{%
  \setstretch{1.0}% single spacing
  \parindent 1em%
  \noindent\hbox to 1.8em{\hss\@makefnmark}#1%
}
\makeatother
