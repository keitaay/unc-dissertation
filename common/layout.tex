% !TeX root = ../main.tex

% Use proper margins
\geometry{letterpaper,left=1in,top=1in,right=1in,bottom=1in,nohead}

% Double-space text
\doublespacing

% Capture the (current) double-spaced baseline as a fixed length.
% Using a fixed length avoids titlesec/setspace interactions later.
\newlength{\docdoubleskip}
\setlength{\docdoubleskip}{\baselineskip}

% Format chapter titles:
\titleformat{\chapter}[block]
  {\centering\bfseries\singlespacing} % applied to both label and title
  {\MakeUppercase{\chaptertitlename\ \thechapter:}}
  {0.5em}                           % separation between label and title
  {\MakeUppercase}                  % ensure title is still uppercased
  [\thispagestyle{plain}\vspace{0.125\docdoubleskip}] % fixed double-space gap after title
\pagestyle{plain}

% Make chapter entries match the style of chapter titles
\renewcommand{\cftchappresnum}{CHAPTER~}
\renewcommand{\cftchapaftersnum}{:\space}
\renewcommand{\cftchapfont}{\normalfont\setstretch{1}} % single-space wrapped lines
\renewcommand{\cftchappagefont}{\normalfont\setstretch{1}} % single-space wrapped lines

% Ensure chapter entries are aligned with other top-level TOC items
\setlength{\cftchapindent}{0pt}

% Make the chapter label box wide enough so the title never overlaps it
\newlength{\tocchapnumwidth}
\settowidth{\tocchapnumwidth}{\normalfont CHAPTER~88:\space}
\addtolength{\tocchapnumwidth}{0.25em} % small safety padding
\setlength{\cftchapnumwidth}{\tocchapnumwidth}

% Extend top margins to 2in + double-space line break after heading
\titlespacing*{\chapter}{0pt}{0.62in}{0pt}

% Keep section/subsection spacing at a single double-spaced line.
% Extra vertical space here stacks with double spacing and looks too large.
\titlespacing*{\section}{0pt}{0pt}{0pt}
\titlespacing*{\subsection}{0pt}{0pt}{0pt}

% Indent paragraphs four spaces throughout the document
\setlength{\parindent}{4ex}

% Tweak spacing of paragraph labels
\titlespacing{\paragraph}{0in}{0.08in}{0.07in}

% Keep text/footnote blocks anchored to the page bottom consistently
\flushbottom

% Place footer page numbers slightly past 0.5in from the bottom margin
% so that the bottom of page numbers align as close to 0.5in from the
% bottom of the page as possible (i.e. removing descender space)
\setlength{\footskip}{0.51in}

% We don't want abandoned words
\clubpenalty=10000 
\widowpenalty=10000

% Prevent awkward hyphenations
\hyphenation{Raj-kumar}

% Ensure superscripts and subscripts are only slightly smaller
% compared to the font size of the main text. Note that the
% arguments define the base, normal text, subscript, and
% superscript font sizes, respectively.
\DeclareMathSizes{12}{12}{10}{10}
