% Change the header and margins for the title, abstract,
% dedication, acknowledgment, and preface pages (i.e. section-l
% ike pages that precede in, and are not included in, the table
% of contents). This will ensure that:
%
% - Text start exactly 2in from the top of the page, overriding
%   TeX's implicit first-line top offset of 1em via "topskip"
%
% - Text is single-spaced
%
% Use this like:
% \pretocheader{\bfseries Title}
\newcommand{\pretocheader}[1]{%
  \par % end any prior paragraph
  \begingroup
    \parskip=0pt%
    \parindent=0pt%
    \singlespacing%
    \setlength{\topskip}{0pt}%
    \vspace*{1in}%
    \noindent\vbox{%
      \hsize=\textwidth
      \centering\normalfont #1\par
    }\par%
    \vspace{\baselineskip}%
  \endgroup
}

% Flush words right at end of paragraph, per:
% http://tex.stackexchange.com/questions/16330/hfill-after-linebreak
\newcommand\rightparend[1]{{%
	\unskip\nobreak\hfil\penalty50
	\hskip2em\hbox{}\nobreak\hfil\textbf{#1}%
	\parfillskip=0pt \finalhyphendemerits=0 \par}
}
